-- Tabel: user_profiles (data onboarding & preferensi)
CREATE TABLE user_profiles (
  id                  BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id             BIGINT UNSIGNED      NOT NULL UNIQUE,
  sex                 ENUM('male','female') NOT NULL,
  birth_date          DATE                 NULL,
  experience_level    ENUM('beginner','intermediate','advanced') NOT NULL DEFAULT 'beginner',
  has_competed        BOOLEAN              NOT NULL DEFAULT FALSE,
  training_days_week  TINYINT UNSIGNED     NOT NULL DEFAULT 3,
  primary_focus       ENUM('sbd','squat','bench','deadlift') NOT NULL DEFAULT 'sbd',
  equipment_type      ENUM('full','home','minimal') NOT NULL DEFAULT 'full',
  onboarding_done     BOOLEAN              NOT NULL DEFAULT FALSE,
  created_at          DATETIME             NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at          DATETIME             NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabel: user_goals (target per kategori)
CREATE TABLE user_goals (
  id              BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id         BIGINT UNSIGNED  NOT NULL,
  goal_type       ENUM('squat_1rm','bench_1rm','deadlift_1rm','body_weight','competition') NOT NULL,
  target_value    DECIMAL(8,2)     NOT NULL,   -- kg untuk lift/bw, 0 untuk kompetisi
  target_date     DATE             NULL,
  competition_name VARCHAR(200)    NULL,        -- hanya untuk goal_type=competition
  federation      VARCHAR(100)     NULL,        -- IPF, USAPL, dll
  is_achieved     BOOLEAN          NOT NULL DEFAULT FALSE,
  achieved_at     DATE             NULL,
  notes           TEXT             NULL,
  created_at      DATETIME         NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at      DATETIME         NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabel: recommendations (cache rekomendasi rule-based + AI)
CREATE TABLE recommendations (
  id              BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id         BIGINT UNSIGNED  NOT NULL,
  source          ENUM('rule','ai') NOT NULL,
  category        ENUM('volume','recovery','weakness','peaking','general') NOT NULL,
  title           VARCHAR(200)     NOT NULL,
  body            TEXT             NOT NULL,
  is_read         BOOLEAN          NOT NULL DEFAULT FALSE,
  generated_at    DATETIME         NOT NULL DEFAULT CURRENT_TIMESTAMP,
  expires_at      DATETIME         NULL,       -- null = tidak expire
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabel: notifications (in-app notification log)
CREATE TABLE notifications (
  id              BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id         BIGINT UNSIGNED  NOT NULL,
  type            ENUM('reminder_recovery','reminder_lift','goal_achieved',
                       'goal_near','competition_countdown','general') NOT NULL,
  title           VARCHAR(200)     NOT NULL,
  message         TEXT             NOT NULL,
  is_read         BOOLEAN          NOT NULL DEFAULT FALSE,
  sent_at         DATETIME         NOT NULL DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabel: notification_preferences (preferensi reminder per user)
CREATE TABLE notification_preferences (
  id                      BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id                 BIGINT UNSIGNED NOT NULL UNIQUE,
  reminder_recovery       BOOLEAN         NOT NULL DEFAULT TRUE,
  reminder_recovery_time  TIME            NOT NULL DEFAULT '20:00:00',
  reminder_lift           BOOLEAN         NOT NULL DEFAULT TRUE,
  reminder_lift_days      VARCHAR(20)     NOT NULL DEFAULT '1,3,5', -- hari (1=Senin..7=Minggu)
  telegram_chat_id        VARCHAR(100)    NULL,    -- opsional Telegram integration
  created_at              DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at              DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tabel: user_dashboard_config (widget preference)
CREATE TABLE user_dashboard_configs (
  id              BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id         BIGINT UNSIGNED NOT NULL UNIQUE,
  widgets         JSON            NOT NULL,
  -- Contoh nilai JSON:
  -- {
  --   "pr_summary": true,
  --   "weekly_volume": true,
  --   "recovery_score": true,
  --   "goal_progress": true,
  --   "recommendation": true,
  --   "wilks_score": false,
  --   "weight_class": false,
  --   "lift_ratio": true
  -- }
  updated_at      DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

BAGIAN 3: LOGIKA BISNIS
================================================================================

--- 3A. LOGIKA GOAL PROGRESS ---

Untuk setiap goal_type, current_value diambil dari:
  squat_1rm     → estimasi 1RM squat terbaru dari lift_records (Epley)
  bench_1rm     → estimasi 1RM bench terbaru dari lift_records
  deadlift_1rm  → estimasi 1RM deadlift terbaru dari lift_records
  body_weight   → entri terbaru dari body_weight_logs
  competition   → tidak ada current_value, tampilkan countdown hari saja

estimated_date (untuk lift goals):
  1. Hitung rate_of_progress = (current_1rm - oldest_1rm_30days) / 30 (kg/hari)
  2. Jika rate_of_progress <= 0 → estimated_date = null (tidak bisa estimasi)
  3. days_needed = (target_value - current_value) / rate_of_progress
  4. estimated_date = today + days_needed

Auto-achieve detection (jalankan di cron job harian):
  Cek semua goal is_achieved = false
  Jika current_value >= target_value → set is_achieved = true, achieved_at = today
  Insert notifikasi type=goal_achieved

--- 3B. RULE-BASED RECOMMENDATION ENGINE ---

Jalankan saat: user trigger POST /recommendations/generate
Input: baca semua data user dari DB (lift_records, recovery_logs, body_weight_logs, user_goals)

Aturan (cek berurutan, bisa generate lebih dari 1 rekomendasi):

RULE 1 — Recovery buruk:
  Kondisi: avg recovery_score 3 hari terakhir < 50
  Category: recovery
  Title: "Recovery kamu sedang rendah"
  Body: "Rata-rata recovery score kamu 3 hari terakhir di bawah 50.
         Pertimbangkan untuk mengurangi volume latihan minggu ini sebesar 20-30%
         dan prioritaskan tidur minimal 8 jam."

RULE 2 — Volume plateau:
  Kondisi: total_volume_kg minggu ini <= total_volume_kg 2 minggu lalu * 1.02
           DAN recovery_score avg 7 hari > 60
  Category: volume
  Title: "Volume latihan kamu stagnan"
  Body: "Volume minggu ini tidak meningkat signifikan padahal recovery kamu baik.
         Coba tambahkan 1 set di lift utama atau naikkan berat 2.5kg."

RULE 3 — Weakness detection:
  Kondisi: weakness dari /analytics/lift-ratio bukan "balanced"
  Category: weakness
  Title: "Terdeteksi kelemahan di lift [nama lift]"
  Body: "Rasio [lift] kamu berada di luar range ideal. Tambahkan 1-2 sesi
         aksesori khusus [lift] per minggu selama 4-6 minggu ke depan."

RULE 4 — Approaching competition:
  Kondisi: ada goal_type=competition dengan target_date dalam 28 hari ke depan
  Category: peaking
  Title: "Kompetisi [nama] tinggal X hari lagi"
  Body: "Saatnya masuk fase peaking. Kurangi volume 40%, pertahankan intensitas
         tinggi (>85% 1RM), dan mulai latihan attempt selection."

RULE 5 — Goal near:
  Kondisi: ada goal dengan progress_pct >= 90 dan is_achieved = false
  Category: general
  Title: "Goal [nama] hampir tercapai!"
  Body: "Kamu sudah mencapai [X]% dari target [goal]. Jaga konsistensi,
         kamu tinggal [Y] kg lagi dari target."

RULE 6 — Tidak ada data recovery:
  Kondisi: tidak ada recovery_log dalam 3 hari terakhir
  Category: recovery
  Title: "Kamu belum mencatat recovery hari ini"
  Body: "Catat tidur dan stres harianmu agar sistem bisa memberikan
         rekomendasi yang lebih akurat."

Setelah generate:
  - Simpan semua rekomendasi baru ke tabel recommendations
  - Set expires_at = NOW() + 7 hari
  - Hapus rekomendasi lama yang sudah expires_at < NOW() milik user ini

--- 3C. AI RECOMMENDATION (OpenRouter) ---

Endpoint: POST /recommendations/ai-analysis
Rate limit: 1x per 24 jam per user (cek di DB)

Prompt yang dikirim ke OpenRouter:

  System:
  "Kamu adalah coach powerlifting berpengalaman. Berikan analisis singkat
   dan rekomendasi program dalam Bahasa Indonesia. Fokus pada data,
   bukan motivasi umum. Maksimal 200 kata."

  User:
  "Berikut data atlet powerlifting saya selama [N] minggu terakhir:
   - Level: [experience_level]
   - Estimasi 1RM: Squat [X]kg | Bench [Y]kg | Deadlift [Z]kg
   - Volume mingguan rata-rata: [X]kg total
   - Recovery score rata-rata: [X]/100
   - Rata-rata tidur: [X] jam, kualitas [X]/10
   - Rata-rata stres: [X]/10
   - Kelemahan terdeteksi: [weakness]
   - Goal aktif: [list goal + progress]
   - Rekomendasi rule-based saat ini: [list title rekomendasi rule]
   
   Berikan analisis dan 2-3 rekomendasi program spesifik untuk minggu depan."

Model: gunakan "meta-llama/llama-3.1-8b-instruct:free" (OpenRouter free tier)
      atau "google/gemma-2-9b-it:free" sebagai fallback

Simpan response ke tabel recommendations dengan source='ai'

Config OpenRouter di .env:
  OPENROUTER_API_KEY=sk-or-...
  OPENROUTER_BASE_URL=https://openrouter.ai/api/v1

--- 3D. LOGIKA CRON JOB ---

Buat scheduler di Go menggunakan goroutine + time.Ticker
Jalankan setiap jam (setiap 60 menit):

Job 1 — Reminder Recovery:
  Untuk setiap user dengan reminder_recovery = true:
    Cek jam sekarang == reminder_recovery_time (toleransi ±5 menit)
    Cek apakah user belum input recovery hari ini
    Jika ya → insert ke notifications (type=reminder_recovery)
    Jika telegram_chat_id ada → kirim pesan via Telegram Bot API

Job 2 — Reminder Lift:
  Untuk setiap user dengan reminder_lift = true:
    Cek hari ini ada di reminder_lift_days
    Cek user belum input lift_record hari ini
    Jika ya → insert ke notifications (type=reminder_lift)

Job 3 — Auto-achieve Goals:
  Jalankan 1x sehari (jam 00:05)
  Loop semua goal is_achieved=false
  Cek current_value vs target_value
  Update dan insert notifikasi jika tercapai

Job 4 — Cleanup:
  Jalankan 1x sehari (jam 03:00)
  Hapus recommendations yang sudah expires_at < NOW()
  Hapus notifications yang sudah > 90 hari

--- 3E. LOGIKA DEFAULT DASHBOARD CONFIG ---

Jika user belum punya config (GET /dashboard/config return 404 dari DB):
  Return default:
  {
    "pr_summary":       true,
    "weekly_volume":    true,
    "recovery_score":   true,
    "goal_progress":    true,
    "recommendation":   true,
    "wilks_score":      false,
    "weight_class":     false,
    "lift_ratio":       true
  }

BAGIAN 7: CONSTRAINT & CATATAN TEKNIS
================================================================================

1.  Onboarding: setelah login, FE cek GET /onboarding/status → jika
    onboarding_done = false, redirect ke /onboarding sebelum Dashboard
2.  Onboarding boleh di-skip (tombol Skip) → set onboarding_done = true
    dengan data minimal (default values)
3.  OpenRouter: gunakan timeout 30 detik, handle error gracefully —
    jika OpenRouter gagal, return error dengan pesan "Analisis AI sedang
    tidak tersedia, coba lagi nanti" (jangan crash handler)
4.  Scheduler: jalankan di goroutine terpisah di main.go, bukan blocking
    go scheduler.Start(db)
5.  Telegram integration: opsional — hanya kirim jika telegram_chat_id
    tidak null dan TELEGRAM_BOT_TOKEN di-set di env
6.  user_dashboard_configs.widgets disimpan sebagai JSON di MariaDB —
    gunakan GORM dengan tipe datatypes.JSON atau serialisasi manual
7.  Rate limit AI: cek di DB, bukan in-memory (supaya restart server
    tidak reset counter)
8.  Rekomendasi rule-based: jangan insert duplikat — cek apakah rekomendasi
    dengan category yang sama sudah ada dan belum expires dalam 24 jam
9.  FE polling notification: gunakan setInterval di Zustand store,
    clear interval saat user logout
10. Goal estimated_date: jika rate_of_progress <= 0 atau data kurang dari
    7 hari, return null dan tampilkan "Belum cukup data untuk estimasi"
11. Semua slider di onboarding dan preferences: gunakan komponen yang
    konsisten dengan halaman Recovery.tsx yang sudah dibuat
12. Semua response error dari OpenRouter harus di-log di server
    (fmt.Printf atau logger) untuk debugging
